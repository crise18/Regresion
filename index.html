#Definimos los parametros y los datos observados----

n = 10

#Iniciales----

prioritheta <- function(theta){
  
  return(dnorm(theta, mean = 1, sd = sqrt(0.16),log = T))
  
}

Para esta parte notemos lo siguiente \[ p=\frac{e^{\theta}}{1+e^{\theta}} \] Así podemos definir lo que sigue en la función de verosimilitud respecto a \(p\) y \(theta\)

#Hallamos la logverosimilitud----
loglikelihood <- function(theta) {
  p <- exp(theta) / (1 + exp(theta))
  p = min(max(p, 1e-8), 1 - 1e-8)
  if (p <= 0 || p >= 1) return(-Inf)
  return(dbinom(y, size = n, prob = p, log =T))#porque tenemos que y distribuye binomial (n,p)
}

Hacemos el algoritmo para Metropolis-Hastings

#Definimos la posteriori
posterior <- function(theta){
  return(loglikelihood(theta)+prioritheta(theta))
}

MetropolisHas = function(valoresiniciales,iteraciones){
  cadena = numeric(iteraciones + 1)
  cadena[1] = valoresiniciales
  naceptacion=0
  nrechazo=0
  for (i in 1:iteraciones){
    propuesta <- rnorm(1, mean = cadena[i], sd = sqrt(0.16))
     razon <- posterior(propuesta) - posterior(cadena[i])
    
    if (is.nan(razon)) {
      cadena[i + 1] <- cadena[i]
      nrechazo <- nrechazo + 1
    } else if (runif(1) < min(1, exp(razon))) {
      cadena[i + 1] <- propuesta
      naceptacion <- naceptacion + 1
    } else {
      cadena[i + 1] <- cadena[i]
      nrechazo <- nrechazo + 1
    }
  }
  return(list(cadena = cadena, naceptacion = naceptacion, nrechazo = nrechazo))
}
    


#Ejecutamos

# Parámetros verdaderos para simular datos
mu_true <- 3
sigma_sq_true <- 4
n <- 2000 # Tamaño de la muestra

# Hiperparámetros para las priors
mu0 <- 0
tau0_sq <- 10 # Varianza de la prior de mu
a <- 1        # Parámetro de forma de la prior de sigma_sq
b <- 2        # Parámetro de escala de la prior de sigma_sq

n_iter <- 10000 # Número total de iteraciones
n_burnin <- 2000 # Número de iteraciones de calentamiento (burn-in)
n_samples <- n_iter - n_burnin # Número de muestras a guardar

set.seed(123)
x_observed <- rtruncnorm(n, a = 0, b = Inf, mean = mu_true, sd = sqrt(sigma_sq_true))

# Inicialización de parámetros
mu_current <- mean(x_observed) 
sigma_sq_current <- var(x_observed)
# Inicializar variables latentes Z
Z_current <- x_observed

# Almacenamiento para las muestras posteriores
mu_samples <- numeric(n_samples)
sigma_sq_samples <- numeric(n_samples)

# Bucle del Muestreador de Gibbs
for (t in 1:n_iter) {
  
  # 1. Muestrear mu | Z, sigma_sq
  # Calcular parámetros de la condicional N(mu_n, sigma_n_sq)
  precision_n <- n / sigma_sq_current + 1 / tau0_sq
  sigma_n_sq <- 1 / precision_n
  mu_n <- sigma_n_sq * (sum(Z_current) / sigma_sq_current + mu0 / tau0_sq)
  # Muestrear nuevo mu
  mu_current <- rnorm(1, mean = mu_n, sd = sqrt(sigma_n_sq))
  
  # 2. Muestrear sigma_sq | Z, mu
  # Calcular parámetros de la condicional InvGamma(a_n, b_n)
  a_n <- a + n / 2
  b_n <- b + 0.5 * sum((Z_current - mu_current)^2)
  # Muestrear nuevo sigma_sq
  sigma_sq_current <- rinvgamma(1, shape = a_n, scale = b_n) # scale=b en InvGamma(a,b)
  
  sigma_sq_current <- max(sigma_sq_current, 1e-9)
  
  # 3. Muestrear Z | mu, sigma_sq, X
  # Muestrear cada Z_i de N(mu, sigma_sq) truncada en (0, Inf)
  Z_current <- rtruncnorm(n, a = 0, b = Inf, mean = mu_current, sd = sqrt(sigma_sq_current))
  
  # Guardar muestras
  if (t > n_burnin) {
    index <- t - n_burnin
    mu_samples[index] <- mu_current
    sigma_sq_samples[index] <- sigma_sq_current
  }
  
  # Imprimir progreso (opcional)
  if (t %% 1000 == 0) {
    cat("Iteración:", t, "/", n_iter, "\n")
  }
}
